/*根据文件名在FileMap中查找文件如"define.lua"
返回string和错误err*/
func (fm FileMap) FindFile(fn string) (fp string, err error) {
	if len(fn) <= 0 {
		err = errors.New("filename is fault!")
		return
	}
	for _, k := range fm {
		fs, errs := os.Stat(k)
		if errs != nil {
			continue
		}
		if fs.Name() == fn {
			fp = k
			err = nil
			break
		}
	}
	if len(fp) <= 0 {
		err = errors.New("can't find this file")
		return
	}
	return
}
